installation :
1- composer create-project laravel/laravel Project-Name 
2- composer create-project laravel/laravel:^9.0 example-app
3- composer global require laravel/installer
 laravel new example-app
4- laravel new example-app --git

publish :
1- php artisan lang:publish

request :
1- php artisan make:request ProductRequest

run :
5- php artisan serve
6- php artisan serve --port=8001
7- php artisan key:generate
8- php artisan lang:publish

maintenance mode :
1- php artisan down
2- php artisan up
3- php artisan down --refresh=5
4- php artisan down --secret="123omar" 
5- php artisan down --render="folder.folder.file" 

controllers :
1- php artisan make:controller TestController
2- php artisan make:controller Test/TestController
3- php artisan make:controller Test/NewsController
4- php artisan make:controller Test/UserController --invokable
5- php artisan make:controller Test/ProductsController --resource

database :
1- php artisan migrate   => run up method
2- php artisan migrate:status
3- WRONG php artisan make:migration products WRONG
4- php artisan make:migration create_categories_table
5- WRONG php artisan make:migration profiles_table WRONG
6- WRONG php artisan make:migration create_branches WRONG
7- php artisan make:migration "create about us table"
8- php artisan migrate:rollback run down in last batch
9- php artisan migrate:refresh run down and after that run up
10- php artisan migrate:fresh => force drop all tables and remigrate(up) 
11- php artisan migrate:rollback --step=5
12  php artisan migrate:rollback --batch=1
13  php artisan migrate:reset  => invoke down only
14- php artisan migrate --path=database/migrations/2024_02_02_153614_add_name_column_to_categories_table.php
15- php artisan make:migration add_name_column_to_categories_table

models :
1- php artisan make:model ModelName
2- php artisan make:model Branch -m
3- php artisan make:model Branch --migration
4- php artisan make:model Branch -mcsfRr [m->migration,c->controller,s->seeder,f->factory,R->request(create,update),r->resource]
5- php artisan make:model Branch --all
6- php artisan make:model Task -msf

factory :
1- php artisan make:factory FactoryName

stubs:
php artisan stub:publish

seeders:
1- php artisan make:seeder UserSeeder
2- php artisan db:seed
3- php artisan migrate --seed
4- php artisan migrate:fresh --seed
5- php artisan db:seed --class=UserSeeder

routing :
1- php artisan route:list
2- php artisan r:l
3- php artisan r:l --except-vendor
