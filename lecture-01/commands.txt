installation :
1- composer create-project laravel/laravel Project-Name 
2- composer create-project laravel/laravel:^9.0 example-app
3- composer global require laravel/installer
 laravel new example-app
4- laravel new example-app --git

run :
5- php artisan serve
6- php artisan serve --port=8001
7- php artisan key:generate
8- php artisan lang:publish

maintenance mode :
1- php artisan down
2- php artisan up
3- php artisan down --refresh=5
4- php artisan down --secret="123omar" 
5- php artisan down --render="folder.folder.file" 

controllers :
1- php artisan make:controller TestController
2- php artisan make:controller Test/TestController
3- php artisan make:controller Test/NewsController


production :
1- php artisan config:cache
